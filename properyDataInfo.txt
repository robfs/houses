select fullkey, type, atom
from houses, json_tree(houses.data) 
where 
    path in (
        '$.propertyData',
        '$.propertyData.prices'
    ) 
and path != '$.propertyData.
and type != 'array'

-- id - property_number
-- bedrooms - number of beds (int)
-- bathrooms - number of baths (int)
-- soldPropertyType - DETACHED
-- propertySubType - Detached
-- description - full description with some guff
-- propertyPhrase - short description (4 bedroom detached house)
-- shareText - ignore
-- shareDescription - ignore
-- pageTitle - short desc
-- shortDescription - shortest desc
-- prices.primaryPrice - £750,000
